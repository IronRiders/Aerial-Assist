008/**
009 * Represents a Logitech Attack3 joystick. This is the joystick that came with

030    /**

033    public static final Button[] BUTTONS = new Button[] {

039    /**
040     * Creates a new instance of the Logitech Attack3 joystick. This is the
041     * joystick that can with the 2012 KOP.

043     * @param port the port that this joystick is plugged in to

045    public Attack3Joystick(int port) {
046        super(port);

053    /**
054     * Repeatedly call this method to check for a button press and notify the
055     * listeners for those buttons.

067            if(((Boolean)state.get(button)) == null){
068                wasPressed = Boolean.valueOf(!isPressed.booleanValue());
069            }
070            else{
071                wasPressed = ((Boolean)state.get(button));
072            }
073            
074            //if (threads[button.getButtonNumber() - 1] == null) {

076            //        public void run(){
077                        notifyButtonListeners(button, isPressed.booleanValue(), wasPressed.booleanValue());

080            //}
081            
082            //if (threads[button.getButtonNumber() - 1] != null) {
083            //    threads[button.getButtonNumber() - 1].start();
084            //}

094    /**
095     * When you want to have code respond to a button press, create a new
096     * {@link ButtonListener} class that responds to the button being pushed
097     * down, and released.
098     * 
099     * Example:
100     * addButtonListener(Button b, new ButtonListener() {
101     *   public void buttonUp() {
102     *     // Now, what do we do when the button is released?
103     *   }
104     * 
105     *   public void buttonDown() {
106     *     // Now, what do we do when the button is pushed?
107     *   }
108     * });
109     * 
110     * @param button the Button to listen to
111     * @param listener the "listener" that responds to the button being pressed or released
112     */
113    public void addButtonListener(Button button, ButtonListener listener) {
114        if (!buttonListeners.contains(button)) {
115            buttonListeners.put(button, new Vector());
116        }
117        
118        ((Vector)(buttonListeners.get(button))).addElement(listener);
119        
120        System.out.println("Attached Button Listener");
121    }
122    public void addButtonListener(Button[] buttons, ButtonListener listener){
123        for(int i = 0; i < buttons.length; i++){
124            addButtonListener(buttons[i], listener);
125            //System.out.println("Attached Button Listener");
126        }
127    }
128            
129    /**
130     * Let the listeners for a button know that is was pressed. If isPressed == wasPressed,
131     * the listeners will not be notified, since the button did not change since
132     * last time
133     * 
134     * @param button the button that was pressed
135     * @param isPressed whether or down the button is now pressed.
136     * @param wasPressed whether or not the button was already pressed.
137     */
138    private void notifyButtonListeners(Button button, boolean isPressed, boolean wasPressed ) {
139        Vector listeners = ((Vector)buttonListeners.get(button));
140        
141        //check to see if listeners is null meaning that there is not listeneer for that button
142        if(listeners == null)
143            return;
144        
145        // If the button is pressed right now, and is was not already pressed,
146        // let the listeners know that the button is now pressed (down)
147        if (isPressed && !wasPressed) {
148            for (int i = 0;i<listeners.size();i++){
149                ButtonListener bl = (ButtonListener)listeners.elementAt(i);
150                bl.buttonDown(button);
151            }
152        }
153        // If the button is not pressed right now, and is was already pressed,
154        // let the listeners know that the button is now pressed released (up)
155        else if (!isPressed && wasPressed) {
156            for (int i = 0;i<listeners.size();i++){
157                ButtonListener bl = (ButtonListener)listeners.elementAt(i);
158                bl.buttonUp(button);
159            }
160        }
161    }

163    /**
164     * When you want to have code respond to joystick movement, create a new
165     * {@link JoystickListener} class that responds to the joystick moving in
166     * any direction
167     * 
168     * Example:
169     * addJoystickListener(new JoystickListener() {
170     *   public void joystickMoved(double x, double y, double z) {
171     *     // Now, what do we do when the button is released?
172     *   }
173     * });
174     * 
175     * @param listener the "listener" that responds to the joystick moving
176     */
177    public void addJoystickListener(JoystickListener listener){
178        if (listener == null){
179           System.out.println("Listener Nonexistant");
180           return;
181        }
182        
183        joystickListeners.addElement(listener);
184        System.out.println("Attached Joystick Listener");
185    }
186    
187    /**
188     * Let the listeners for the joystick know that is was moved.
189     */
190    private void notifyJoystickListeners(){
191        double x = this.getX();
192        double y = this.getY();
193        double z = this.getZ();
194        z = (-z+1)/2;
195        
196        for(int i = 0; i < joystickListeners.size(); i ++){
197            JoystickListener joystickListener = (JoystickListener)joystickListeners.elementAt(i);
198            if ( joystickListener == null){
199                System.out.println("No Listener at Index "+ i);
200            }
201            joystickListener.joystickMoved(x, y, z);
202        }
203    }
