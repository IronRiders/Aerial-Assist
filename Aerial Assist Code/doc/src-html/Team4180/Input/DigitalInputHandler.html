007package Team4180.Input;
008
009import Team4180.Listeners.SwitchListener;
010import Team4180.Robot;
011import edu.wpi.first.wpilibj.DigitalInput;
012import java.util.Hashtable;
013import java.util.Vector;
014
015/**
016 *
017 * @author ros_haremash
018 */
019public class DigitalInputHandler {
020    
021    private final Hashtable listenersTable;
022    private final Boolean[] stateTable;
023    
024    private final int[] ports;
025    private final DigitalInput[] digitalInputs;
026    
027    //This method assigns the given ports to vaules in a DigitalInput array
028    //Accepts int array or ports in which limit switches will be used
029    public DigitalInputHandler(int[] digitalInputPorts){
030        ports = digitalInputPorts;
031        
032        listenersTable = new Hashtable();
033        stateTable = new Boolean[digitalInputPorts.length];
034       
035        digitalInputs = new DigitalInput[ports.length];
036        for(int i = 0; i < ports.length; i++){
037            digitalInputs[i] = new DigitalInput(ports[i]); 
038            System.out.println(digitalInputs[i].getChannel() + ", " + digitalInputs[i].getAnalogTriggerForRouting() + ", " + digitalInputs[i].getChannelForRouting() + ", " + digitalInputs[i].getModuleForRouting());
039        }
040    }
041    
042    //This method recieves vaules from the robot's DigitalInput/limit switches
043    //Gets vaules from digital input ports
044    public void listen(){
045        //the handler is lstening
046        for(int i = 0; i < ports.length; i++){
047            //System.out.println("is switched was switched");
048            final boolean isSwitched = !digitalInputs[i].get();
049            final boolean wasSwitched;
050            
051            // Was the button pressed last time we checked?
052            if((Boolean)stateTable[i] == null){
053                wasSwitched = !isSwitched;
054                //System.out.println("line 54");
055            }
056            else{
057                wasSwitched = ((Boolean)stateTable[i]).booleanValue();
058                //System.out.println("Line 58");
059            }
060            stateTable[i] = Boolean.valueOf(isSwitched);
061            //notifiying the switch listeners
062            //System.out.println("line 62");
063            //this is where the error is happening
064            try {
065                notifySwitchListeners(ports[i], isSwitched, wasSwitched);
066            } catch(NullPointerException e) {
067                
068            }
069            //System.out.println("line 69");
070        }
071    }
072    
073    //This method assigns vaules to the Hashtable listnersTable, a number port
074    //is assigned as the key of the Vector listeners
075    //Accepts the int portnumber that the switch listener is assigned to and the
076    //SwitchListner listener
077    public void addSwitchListener(int portNumber, SwitchListener listener){
078        Integer numberPort = Integer.valueOf(portNumber);
079        Vector listeners = (Vector) listenersTable.get(numberPort);
080        if(listeners == null) {
081            listeners = new Vector();
082            listenersTable.put(numberPort, listeners);
083        }
084        listeners.addElement(listener);
085        System.out.println("Attached Switch Listener to port " + numberPort);
086    }
087    
088    public void addSwitchListener(int[] portNumbers, SwitchListener listener){
089        for(int i = 0; i < portNumbers.length; i++){
090            addSwitchListener(portNumbers[i], listener);
091        }
092    }
093    
094    private void notifySwitchListeners(int portNumber, boolean isSwitched, boolean wasSwitched) {
095        if (Robot.DEBUG_SWITCH_LISTENER) { System.err.println("Notifying listeners on port " + portNumber); }
096        Vector listeners = ((Vector)listenersTable.get(Integer.valueOf(portNumber)));
097        if (Robot.DEBUG_SWITCH_LISTENER) { System.err.println("Port " + portNumber + " has " + (listeners == null ? 0 : listeners.size()) + " listeners"); }
098        
099        if(listeners == null)
100            return;
101
102        if (Robot.DEBUG_SWITCH_LISTENER) {
103            System.out.println("isSwitched: " + isSwitched);
104            System.out.println("wasSwitched: " + wasSwitched);
105        }
106        
107        for (int i = 0; i < listeners.size(); i++) {
108            SwitchListener sl = (SwitchListener)listeners.elementAt(i);
109            
110            if (isSwitched != wasSwitched) {
111                if (Robot.DEBUG_SWITCH_LISTENER) { 
112                   System.out.println("Calling listener " + i + " for port " + portNumber);
113                }
114                sl.switchState(portNumber, isSwitched);
115            }
116        }
117            for (int i = 0; i < listeners.size(); i++){
118                SwitchListener sl = (SwitchListener)listeners.elementAt(i);
119                if(isSwitched){
120                    sl.switchOn(portNumber);
121                }else{
122                    sl.switchOff(portNumber);
123                }
124            }
125        }
126}
