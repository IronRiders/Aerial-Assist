077    /**
078     * This function is run when the robot is first started up and should be
079     * used for any initialization code.
080     */
081    public void robotInit() {
082        
083        //initialize everything bit
084        joystick1 = new Attack3Joystick(1);
085        joystick2 = new Attack3Joystick(2);
086        digitalInputPorts = new int[] {PRESSURE_LIMIT_PORT, BALL_CAUGHT_PORT,CLOSECLAW_LIMIT_SWITCH, OPENCLAW_LIMIT_SWITCH, RELOAD_LIMIT_SWITCH, DETECT_BALL_PORT, ELEVATION_SWITCH_LISTENER_PORT};
087        digitalInputHandler = new DigitalInputHandler(digitalInputPorts);
088        
089        // driving
090        driving = new Driving();
091        triggerL = new TriggerListener(driving);
092        movementL = new MovementListener(driving);
093 
094        // claw
095        claw = new Claw();
096        pressure = new Pressure();
097        pressureLimitL = new PressureLimitListener(pressure);
098        clawElevationL = new ElevationListener(claw);
099        clawButtonL = new ClawButtonListener(claw,pressure);
100        clawLimitL = new ClawSwitchListener(claw, CLOSECLAW_LIMIT_SWITCH, OPENCLAW_LIMIT_SWITCH);
101        reloadSwitchL = new ReloadSwitchListener(claw,RELOAD_LIMIT_SWITCH);
102        elevationSwitchL = new ElevationSwitchListener(claw,ELEVATION_SWITCH_LISTENER_PORT);
103        ballDetectL = new BallDetectListener(claw, DETECT_BALL_PORT);
104        ballCaughtL = new BallCaughtListener(claw, BALL_CAUGHT_PORT);
105        claw.setClawElevationState(Claw.ElevationState.GOING_UP);
106        
107        //joysticks
108        joystick1.addJoystickListener(movementL);
109        joystick1.addButtonListener(Button.TRIGGER, triggerL);
110        joystick2.addJoystickListener(clawElevationL); 
111        joystick2.addButtonListener(Button.TRIGGER, clawButtonL);
112        joystick2.addButtonListener(Button.TRIGGER,  clawButtonL);
113        joystick2.addButtonListener(Button.BUTTON_2, clawButtonL);
114        joystick2.addButtonListener(Button.BUTTON_3, clawButtonL);
115        joystick2.addButtonListener(Button.BUTTON_6, clawButtonL);
116        joystick2.addButtonListener(Button.BUTTON_4, clawButtonL);
117        joystick2.addButtonListener(Button.BUTTON_5, clawButtonL);
118        joystick2.addButtonListener(Button.BUTTON_6, clawButtonL);//should this listent to fireL? Should it restart the autorelease?
119        joystick2.addButtonListener(Button.BUTTON_7, clawButtonL);
120        joystick2.addButtonListener(Button.BUTTON_8, clawButtonL);
121        joystick2.addButtonListener(Button.BUTTON_9, clawButtonL);
122        joystick2.addButtonListener(Button.BUTTON_10, clawButtonL);
123        joystick2.addButtonListener(Button.BUTTON_11, clawButtonL);
124        
125        //limit switches
126        digitalInputHandler.addSwitchListener(PRESSURE_LIMIT_PORT, pressureLimitL);
127        digitalInputHandler.addSwitchListener(OPENCLAW_LIMIT_SWITCH, clawLimitL);
128        digitalInputHandler.addSwitchListener(CLOSECLAW_LIMIT_SWITCH, clawLimitL);
129        digitalInputHandler.addSwitchListener(ELEVATION_SWITCH_LISTENER_PORT, elevationSwitchL);
130        digitalInputHandler.addSwitchListener(DETECT_BALL_PORT, ballDetectL);
131        digitalInputHandler.addSwitchListener(BALL_CAUGHT_PORT, ballCaughtL);
132    }
