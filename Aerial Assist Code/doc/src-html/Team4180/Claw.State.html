006package Team4180;
007
008import edu.wpi.first.wpilibj.Jaguar;
009import edu.wpi.first.wpilibj.Servo;
010import edu.wpi.first.wpilibj.Talon;
011import edu.wpi.first.wpilibj.Timer;
012
013/**
014 *
015 * @author ros_aljacobson
016 */
017public class Claw {
018
019    private Talon openClose;
020    private Jaguar raiseLower;
021    //private Talon pullBackPiston;
022    private Talon drumMotor;
023    private Servo ratchet;
024    private Servo clutch;
025    private double yValueNow;
026    private double zValueNow = 1;
027    private String clawState;
028    private String clawElevationState;
029    
030
031    //These are which ports each motor is on the robot
032    public static final int TEMP_PORT = 10; //if hardware has not given us a port yet then TEMP_PORT is used
033    public static final int CLAW_OPEN_CLOSE_PORT = 6;
034    public static final int CLAW_RAISE_LOWER_PORT = 5;
035    public static final int RATCHET_PORT = 10;
036    public static final int DRUM_PORT = 4;
037    public static final int CLUTCH_SERVO_PORT = 7;//not temp
038
039    public static final double DISENGAGE_CLUTCH_ANGLE = 70; // -20
040    public static final double ENGAGE_CLUTCH_ANGLE = 180; // -100
041    public static final double DISENGAGE_RATCHET_ANGLE = 100;
042    public static final double ENGAGE_RATCHET_ANGLE = 0;
043    
044    //controls the speed of the talons for the claw and drum
045    public final double CLOSE_CLAW_VALUE = -1;
046    public final double OPEN_CLAW_VALUE = 1;
047    public final double DRUM_FORWARD_SPEED = 1;
048    public final double DRUM_BACKWARDS_SPEED = -1;
049
050    public Servo getClutch() {
051        return clutch;
052    }
053
054    public final class State {
055        public static final String OPEN = "open";
056        public static final String OPENING = "opening";
057        public static final String CLOSED = "closed";
058        public static final String CLOSING = "closing";
059    }
060    
061    public final class ElevationState
062    {
063        public static final String GOING_UP = "going up";
064        public static final String GOING_DOWN = "going down";
065        public static final String DOWN = "down";
066    }
067
068    public Claw() {
069        //This assigns the ports to each of the seperate motors (talons and servos.
070        //each motor will get input from the robot through its assigned port.
071        openClose = new Talon(CLAW_OPEN_CLOSE_PORT);
072        raiseLower = new Jaguar(CLAW_RAISE_LOWER_PORT);
073        //pullBackPiston = new Talon(PULL_BACK_PISTON_PORT);
074        drumMotor = new Talon(DRUM_PORT);
075        ratchet = new Servo(RATCHET_PORT);
076        
077        clutch = new Servo(CLUTCH_SERVO_PORT);
078        yValueNow = 0;
079        clawState = State.CLOSED;
080        clawElevationState = ElevationState.GOING_UP;
081    }
082
083    //This method opens the claw. It takes in a double from a button listener(ClawOpenListener).
084    public void openClaw() {
085        Robot.log(Robot.DEBUG_CLAW, "Opening claw...");
086        if (!clawState.equals(State.OPEN)) {
087            openClose.set(OPEN_CLAW_VALUE);
088            setClawState(State.OPENING);
089            System.out.println("opening claw");
090        }
091        else {
092            Robot.log(Robot.DEBUG_CLAW, "Cannot open claw more");
093        }
094    }
095    
096    //This method closes the claw. It takes a negative double from the ClawCloseListener.
097    public void closeClaw() {
098        Robot.log(Robot.DEBUG_CLAW, "Closing claw...");
099        if (!clawState.equals(State.CLOSED)) {
100            openClose.set(CLOSE_CLAW_VALUE);
101            setClawState(State.CLOSING);
102        }
103        else {
104            Robot.log(Robot.DEBUG_CLAW, "Cannot close claw more");
105        }
106    }
107    
108    public void setClawState(String state) {
109        clawState = state;
110        Robot.log(Robot.DEBUG_CLAW, "Claw state: " + state);
111    }
112    
113    public void setClawElevationState(String state)
114    {
115        String old = clawElevationState;
116        clawElevationState = state;
117        if (old != null && !old.equals(state))
118            Robot.log(Robot.DEBUG_CLAW, "Claw elevation state: " + state);;
119    }
120
121    public String getClawState() {
122        return clawState;
123    }
124    
125    public String getClawElevationState()
126    {
127        return clawElevationState;
128    }
129
130    public void stopClaw() {
131        double old = openClose.get();
132        openClose.set(0);
133        if (old != 0)
134            Robot.log(Robot.DEBUG_CLAW, "Claw stopped");
135    }
136
137    public void updateY(double y) {
138        yValueNow = y;
139    }
140
141    public void updateZ(double z) {
142        zValueNow = z;
143    }
144
145    public void clawElevation() {
146        double newY = ((double) (int) (yValueNow * 100)) / 100;
147        raiseLower.set(newY);
148        if(newY > 0) {
149            setClawElevationState(Claw.ElevationState.GOING_UP);
150        }
151        else if(newY < 0 && !getClawElevationState().equals(Claw.ElevationState.DOWN)) {
152            setClawElevationState(Claw.ElevationState.GOING_DOWN);
153        }
154    }
155
156    public void stopElevation() {
157        raiseLower.set(0);
158        Robot.log(Robot.DEBUG_CLAW, "Claw elevation stopped");
159    }
160
161    //This method releases the ratchet and releases the spring system to let the robot shoot the ball.
162    //When the cocking servo reaches the angle (100 degrees) that is equal to the disengae clutch angle
163    //then the disengage ratchet method is called.
164    public void fire() {
165        //100 --> class constant
166        disengageRatchet();
167        //if (clutch.get() == DISENGAGE_CLUTCH_ANGLE) {
168          //  disengageRatchet();
169        //}
170        Timer.delay(1);
171        reload();
172    }
173
174    //Automatically reloads the shooting mechanism
175    public void reload() {
176        engageRatchet();
177        engageClutch();
178        drumBackwards();
179    }
180
181    public void stopReload() {
182        drumStop();
183        disengageClutch();
184    }
185
186    public void engageClutch() {
187        clutch.setAngle(ENGAGE_CLUTCH_ANGLE);
188        Robot.log(Robot.DEBUG_CLAW, "Clutch engaged.  Angle: " + clutch.getAngle());
189    }
190
191    public void disengageClutch() {
192        clutch.setAngle(DISENGAGE_CLUTCH_ANGLE);
193        Robot.log(Robot.DEBUG_CLAW, "Clutch disengaged. Angle: " + clutch.getAngle());
194    }
195
196    public void drumForwards() {
197        drumMotor.set(DRUM_FORWARD_SPEED);
198        Robot.log(Robot.DEBUG_CLAW, "Moving the drum forwards");
199    }
200
201    public void drumBackwards() { 
202        drumMotor.set(DRUM_BACKWARDS_SPEED);
203        Robot.log(Robot.DEBUG_CLAW, "Moving the drum forwards");
204    }
205
206    public void drumStop() {
207        drumMotor.stopMotor();
208        Robot.log(Robot.DEBUG_CLAW, "Drum stopped");
209    }
210
211    //This method disengages the ratchet to allow fireing. 
212    //This method is called upon in the method fire.
213    public void disengageRatchet() {
214        ratchet.set(DISENGAGE_RATCHET_ANGLE);
215        Robot.log(Robot.DEBUG_CLAW, "Ratchet disengaged");
216    }
217
218    //This method engages the ratchet to hold the system in place after the reload process.
219    //The method is used when called upon by the auto reload.
220    public void engageRatchet() {
221        ratchet.set(ENGAGE_RATCHET_ANGLE);
222        Robot.log(Robot.DEBUG_CLAW, "Ratchet engaged");
223    }
224    
225    public void increaseClutchAngle() {
226        clutch.setAngle(clutch.getAngle()+1);
227        Robot.log(Robot.DEBUG_CLAW, "Increased clutch angle to " + clutch.getAngle());
228    }
229    
230    public void decreaseClutchAngle() {
231        clutch.setAngle(clutch.getAngle()-1);
232        Robot.log(Robot.DEBUG_CLAW, "Decreased clutch angle to " + clutch.getAngle());
233    }
234    
235    public void openClawA(double speed)
236    {
237        openClose.set(speed);
238        Robot.log(Robot.DEBUG_CLAW, "Opening claw...");
239        setClawState(State.OPENING);
240    }
241    
242    public void closeClawA(double speed)
243    {
244        openClose.set(-speed);
245        Robot.log(Robot.DEBUG_CLAW, "Closing claw...");
246        setClawState(State.CLOSING);
247    }
248}
