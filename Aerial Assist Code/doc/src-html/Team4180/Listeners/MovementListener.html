007package Team4180.Listeners;
008import Team4180.Driving;
009
010/**
011 *
012 * @author ros_kecooper
013 */
014public class MovementListener implements JoystickListener {
015    private Driving driving;
016    
017    //Runs driving class 
018    //Takes in driving
019    public MovementListener (Driving driving){
020        this.driving = driving;
021    }
022    //Determines if the joystick has moved and updates    
023    //Takes in the coordinates of joystick position
024    public void joystickMoved(double x, double y, double z){
025        System.out.println("x = " + x);
026        System.out.println("y = " + y);
027        System.out.println("z = " + z);
028        driving.updateX(x);
029        driving.updateY(y);
030        driving.updateZ(z);
031        driving.reCalcRightVelocity();
032    }
033}
