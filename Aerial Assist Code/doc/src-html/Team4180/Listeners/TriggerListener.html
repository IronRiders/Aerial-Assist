007package Team4180.Listeners;
008
009import Team4180.Driving;
010import Team4180.Input.Attack3Joystick;
011import edu.wpi.first.wpilibj.Servo;
012/**
013 *
014 * @author ros_diwang
015 */
016public class TriggerListener implements ButtonListener{
017    private Driving driving;
018    
019    //Runs driving class 
020    //Takes in driving
021    public TriggerListener(Driving driving){
022        this.driving = driving;
023    }
024    //This starts the forward movement of the robot if the trigger is released
025    //Takes a button, number 1 (trigger)
026    public void buttonUp(Attack3Joystick.Button button){
027        driving.driveStraight(false);
028    }
029    //This stops the forward movement of the robot if the trigger is released
030    //Takes a button, number 1 (trigger)
031    public void buttonDown(Attack3Joystick.Button button){
032        driving.driveStraight(true);   
033    }    
034    
035   
036
037}
