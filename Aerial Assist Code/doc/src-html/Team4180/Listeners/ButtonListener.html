001package Team4180.Listeners;
002
003import Team4180.Input.Attack3Joystick.Button;
004
005/**
006 * Interface describing actions that a ButtonListener must react to.
007 */
008public interface ButtonListener {
009        
010   /**
011    * This method will be called when the button is released (up).
012    * 
013    * @param button the button that was pushed
014    */
015    public void buttonUp(Button button);
016        
017    /**
018     * This method will be called when the button is pressed (down).
019     * 
020     * @param button the button that was pushed
021     */
022    public void buttonDown(Button button);
023}
