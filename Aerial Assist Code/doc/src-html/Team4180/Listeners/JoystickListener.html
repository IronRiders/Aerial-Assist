001package Team4180.Listeners;
002
003/**
004 * Interface describing actions that a JoystickListener must react to.
005 */
006public interface JoystickListener {
007    
008    /**
009     * This method will be called whenever the joystick is moved
010     * 
011     * @param x the X (+up / -down) position of the joystick
012     * @param y the Y (+left / -right) position of the joystick
013     * @param z the Z position of the joystick
014     */
015    public void joystickMoved(double x, double y, double z);
016}
