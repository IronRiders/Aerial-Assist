007package Team4180;
008
009import edu.wpi.first.wpilibj.Relay;
010import edu.wpi.first.wpilibj.Solenoid;
011
012/**
013 *
014 * @author Aaron Jacobson
015 */
016public class Pressure {
017    private static final int TEMP_PORT = 10;
018    private int RELAY_PORT = 1;
019    private Relay pressureRelay;
020    private int SHOOT_PORT = 1;
021    private Solenoid shoot;
022    public Pressure()
023    {
024        pressureRelay = new Relay(RELAY_PORT);
025        shoot = new Solenoid(SHOOT_PORT);
026        
027    }
028    
029    public void relayReverse(){
030        pressureRelay.set(Relay.Value.kReverse);
031    }
032    public void relayOn()
033    {
034        pressureRelay.set(Relay.Value.kForward);
035    }
036    
037    public void relayOff() {
038        pressureRelay.set(Relay.Value.kOn);
039    }
040    
041    public int getPressureRelayPort()
042    {
043        return RELAY_PORT;
044    }
045    
046    public void solenoidOn(){
047        shoot.set(true);
048    }
049    
050    public void solenoidOff(){
051        shoot.set(false);
052    }
053}
